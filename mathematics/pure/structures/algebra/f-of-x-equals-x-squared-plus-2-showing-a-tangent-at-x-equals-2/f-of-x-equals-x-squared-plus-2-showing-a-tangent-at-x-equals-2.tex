\documentclass{standalone}
\usepackage{tikz}
\usepackage{pgfplots}
\usetikzlibrary{arrows.meta}

\begin{document}

\resizebox{8cm}{8cm}{%
    \begin{tikzpicture}
    \begin{axis}[
        title={},
        axis lines=middle,
        axis line style={Stealth-Stealth,very thick},
        xlabel=$x$,
        ylabel={$f(x)$},
        xmin=-6.5,xmax=6.5,ymin=-6.5,ymax=6.5,
        xtick distance=2,
        ytick distance=2,
        grid=major,
        grid style={thin,densely dotted,black!20},
        % Bottom center
        legend style={
            at={(0.5,-0.1)},
            anchor=north,
            cells={anchor=west}
        }
    ]
    \addplot[
        domain= -7:7,
        samples=50,
        color=red
        ]
        {x^2+2};
    % Add tangent line at x = 2
    \addplot[
        domain= 1:7,
        samples=50,
        color=blue
        ]
        {4*x-2};
    \addlegendentry{$f(x) = x^2+2$}
    \end{axis}
    \end{tikzpicture}
}

\end{document}
