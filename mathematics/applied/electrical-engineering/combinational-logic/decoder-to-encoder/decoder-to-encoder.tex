\documentclass[border=3mm]{standalone}
\usepackage[siunitx, RPvoltages]{circuitikz}
\usetikzlibrary{arrows, shapes.gates.logic.US, shapes.gates.logic.IEC, calc}

\tikzset{
    my3to8/.style={
        muxdemux, muxdemux def={
            Lh=8, Rh=8, w=6,  % both left and right size =8 units, wide top
            NT=0,             % none top
            NB=0,             % none bottom
            NL=1,             % 1 pins on the left
            NR=8,             % 8 pins on the right
        },
        % draw only left pins={1},
    },
    my8to3/.style={
        muxdemux, muxdemux def={
            Lh=8, Rh=8, w=6,  % both left and right size =8 units, wide top
            NT=0,             % none top
            NB=0,             % none bottom
            NL=8,             % 8 pins on the left
            NR=1,             % 1 pins on the right
        },
        % draw only left pins={1},
    },
}

\begin{document}

\resizebox{10cm}{!}{

    \begin{circuitikz}[scale=1.0, transform shape]
        
        % DECODER *********************************************************************

            \node[my3to8]                         (DECODE)   at (0,0)                            {};
            %LEFT
            \node[right, font=\tiny\ttfamily]     (DPIN0)    at ($(DECODE.blpin 1) + (0, 0)$)     {\normalsize $[2\:0]in$};
            \coordinate[]                         (DIN0)     at ($(DECODE.blpin 1) + (-1, 0)$)    {};
            %RIGHT
            \foreach \i [count=\iz from 0] in {1,...,8} {
                \node[left, font=\tiny\ttfamily]  (DPOUT\iz) at ($(DECODE.brpin \i) + (0, 0)$)    {\normalsize $out[\iz]$};
                \coordinate[]                     (DOUT\iz)  at ($(DECODE.brpin \i) + (1, 0)$)    {};
            }
            %TOP
            \node[]                               (NAME)    at ($(DECODE) + (0, 3.0)$)           {\Large \textbf {DECODER 3-8}};
            %CONNECT
            \draw[] (DECODE.blpin 1) -- (DIN0);
            \foreach \i [count=\iz from 0] in {1,...,8} {
                \draw[] (DECODE.brpin \i) -- (DOUT\iz);
            }
            
            % INPUTS
            \node[] (INPUT0)  at ($(DIN0) + (-.3, 0)$)  {\normalsize $010$};
        
        % ENCODER *********************************************************************

            \node[my8to3]                         (ENCODE) at ($(DECODE) + (6, 0)$)              {};
            %LEFT
            \foreach \i [count=\iz from 0] in {1,...,8} {
                \node[right, font=\tiny\ttfamily] (EPIN\iz) at ($(ENCODE.blpin \i) + (0, 0)$)     {\normalsize $in[\iz]$};
                \coordinate[]                     (EIN\iz)  at ($(ENCODE.blpin \i) + (-1, 0)$)    {};
            }
            %RIGHT
            \node[left, font=\tiny\ttfamily]      (EPOUT0)  at ($(ENCODE.brpin 1) + (0, 0)$)      {\normalsize $[2\:0]out$};
            \coordinate[]                         (EOUT0)   at ($(ENCODE.brpin 1) + (1, 0)$)      {};        
            %TOP
            \node[]                               (NAME)   at ($(ENCODE) + (0, 3.0)$)            {\Large \textbf {ENCODER 8-3}};
            %CONNECT
            \foreach \i [count=\iz from 0] in {1,...,8} {
                \draw[] (ENCODE.blpin \i) -- (EIN\iz);
            }
            \draw[] (ENCODE.brpin 1) -- (EOUT0);

            % OUTPUTS
            \node[] (INPUT0)  at ($(EOUT0) + (.3, 0)$)  {\normalsize $010$};

        % CONNECT **********************************************************************

        \foreach \i [count=\iz from 0] in {1,...,8} {
            \draw[] (DOUT\iz) -- (EIN\iz);
        }

    \end{circuitikz}

}

\end{document}
